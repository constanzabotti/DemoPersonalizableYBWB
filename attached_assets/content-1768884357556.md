```
import sqlite3

def crear_base_de_datos():
    conexion = sqlite3.connect('fitness_app.db')
    cursor = conexion.cursor()

    # Tabla de Usuarios (Entrenadores y Alumnos)
    cursor.execute('''
        CREATE TABLE IF NOT EXISTS usuarios (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            nombre TEXT NOT NULL,
            rol TEXT CHECK(rol IN ('entrenador', 'alumno')),
            is_premium BOOLEAN DEFAULT 0,
            puntos_prestigio INTEGER DEFAULT 0
        )
    ''')

    # Tabla de Check-in diario (Tu valor diferencial)
    cursor.execute('''
        CREATE TABLE IF NOT EXISTS checkins (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            alumno_id INTEGER,
            fecha DATE DEFAULT CURRENT_DATE,
            horas_sueno INTEGER,
            nivel_estres INTEGER,
            energia INTEGER,
            FOREIGN KEY(alumno_id) REFERENCES usuarios(id)
        )
    ''')

    # Tabla de Entrenamientos
    cursor.execute('''
        CREATE TABLE IF NOT EXISTS entrenamientos (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            alumno_id INTEGER,
            ejercicio TEXT,
            series INTEGER,
            reps INTEGER,
            completado BOOLEAN DEFAULT 0,
            FOREIGN KEY(alumno_id) REFERENCES usuarios(id)
        )
    ''')

    conexion.commit()
    conexion.close()
    print("Base de datos e infraestructura de monetización listas.")

if __name__ == "__main__":
```

```
import sqlite3

def crear_base_de_datos():
    conexion = sqlite3.connect('fitness_app.db')
    cursor = conexion.cursor()

    # 1. USUARIOS (Con lógica Premium y Puntos)
    cursor.execute('''
        CREATE TABLE IF NOT EXISTS usuarios (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            nombre TEXT NOT NULL,
            rol TEXT CHECK(rol IN ('entrenador', 'alumno')),
            is_premium BOOLEAN DEFAULT 0, -- $10/mes para entrenadores
            puntos_motivacion INTEGER DEFAULT 0, -- Para alumnos
            metodo_pago_preferido TEXT -- Venmo, Zelle, Cash, etc.
        )
    ''')

    # 2. PRESUPUESTOS (El entrenador arma la oferta)
    cursor.execute('''
        CREATE TABLE IF NOT EXISTS presupuestos (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            entrenador_id INTEGER,
            alumno_id INTEGER,
            tipo_plan TEXT CHECK(tipo_plan IN ('clase', 'semanal', 'mensual')),
            monto DECIMAL(10, 2),
            estado TEXT DEFAULT 'pendiente', -- pendiente, pagado
            FOREIGN KEY(entrenador_id) REFERENCES usuarios(id),
            FOREIGN KEY(alumno_id) REFERENCES usuarios(id)
        )
    ''')

    # 3. PAGOS (Registro de transacciones)
    cursor.execute('''
        CREATE TABLE IF NOT EXISTS pagos (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            presupuesto_id INTEGER,
            metodo_utilizado TEXT, -- App, Cash, Venmo, Zelle
            fecha_pago DATETIME DEFAULT CURRENT_TIMESTAMP,
            confirmado_por_entrenador BOOLEAN DEFAULT 0,
            FOREIGN KEY(presupuesto_id) REFERENCES presupuestos(id)
        )
    ''')

    # 4. DESAFÍOS Y RECOMPENSAS (Para ganar acceso a contenido)
    cursor.execute('''
        CREATE TABLE IF NOT EXISTS recompensas (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            nombre_premio TEXT,
            puntos_requeridos INTEGER,
            contenido_url TEXT -- Link al PDF o rutina secreta
        )
    ''')

    # 5. TABLA DE CHECK-IN Y ENTRENAMIENTOS (Se mantienen del anterior)
    cursor.execute('CREATE TABLE IF NOT EXISTS checkins (id INTEGER PRIMARY KEY, alumno_id INTEGER, fecha DATE, horas_sueno INTEGER, nivel_estres INTEGER, energia INTEGER)')

    conexion.commit()
    conexion.close()
    print("Base de datos con Sistema Financiero y Gamificación lista.")

if __name__ == "__main__":
    crear_base_de_datos()
```
