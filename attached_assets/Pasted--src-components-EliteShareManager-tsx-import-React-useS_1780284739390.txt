// src/components/EliteShareManager.tsx
import React, { useState, useEffect } from 'react';

// Detecta automáticamente el link actual (sea el Replit original o tu Remix)
const LINK_A_COMPARTIR = typeof window !== 'undefined' ? window.location.href : 'https://tu-app.replit.dev/';

interface ShareProps {
  isOpen: boolean;
  onClose: () => void;
}

const EliteShareManager: React.FC<ShareProps> = ({ isOpen, onClose }) => {
  const [estadoCopiado, setEstadoCopiado] = useState<string | null>(null);

  // Intento de compartido nativo (Plan A)
  const ejecutarCompartidoNativo = async () => {
    if (navigator.share && navigator.canShare && navigator.canShare({ url: LINK_A_COMPARTIR })) {
      try {
        await navigator.share({
          title: 'You Better Work B*tch! | Elite SaaS',
          text: 'Mira esta plataforma de entrenamiento y gamificación premium.',
          url: LINK_A_COMPARTIR,
        });
        onClose(); // Si funciona, cerramos el modal
        return;
      } catch (error) {
        if ((error as Error).name !== 'AbortError') {
          console.error('Error al compartir nativo:', error);
        }
      }
    }
    // Si el navegador no lo soporta, el código no se rompe y se queda abierto el Plan B (Modal de Copiado)
  };

  // Copiar al portapapeles (Plan B)
  const handleCopiarEnlace = () => {
    navigator.clipboard.writeText(LINK_A_COMPARTIR).then(
      () => {
        setEstadoCopiado('¡Enlace Copiado! 🚀');
        setTimeout(() => {
          setEstadoCopiado(null);
          onClose(); // Cierra automáticamente después de copiar
        }, 1500);
      },
      (err) => {
        setEstadoCopiado('Error al copiar. Intenta manualmente.');
        console.error('Error en portapapeles: ', err);
      }
    );
  };

  useEffect(() => {
    if (isOpen) {
      ejecutarCompartidoNativo();
    }
  }, [isOpen]);

  if (!isOpen) return null;

  // Interfaz de respaldo estética neón (Plan B visible)
  return (
    <div className="fixed inset-0 bg-black/80 backdrop-blur-md flex items-center justify-center p-4 z-[99]">
      <div className="bg-[#0a0a0a] border border-[#39FF14]/30 rounded-2xl p-8 max-w-sm w-full text-center shadow-[0_0_50px_rgba(57,255,20,0.1)]">
        
        <div className="w-16 h-16 bg-[#39FF14]/10 text-[#39FF14] rounded-full flex items-center justify-center mx-auto mb-4 border border-[#39FF14]/20">
          <svg xmlns="http://www.w3.org/2000/svg" className="h-8 w-8" fill="none" viewBox="0 0 24 24" stroke="currentColor">
            <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M8 16H6a2 2 0 01-2-2V6a2 2 0 012-2h8a2 2 0 012 2v2m-6 12h8a2 2 0 002-2v-8a2 2 0 00-2-2h-8a2 2 0 00-2 2v8a2 2 0 002 2z" />
          </svg>
        </div>

        <h3 className="text-xl font-bold text-white uppercase tracking-tight mb-2">
          {estadoCopiado || 'Comparte la Plataforma'}
        </h3>
        <p className="text-gray-400 text-sm mb-6">
          {estadoCopiado ? 'Ya puedes pegarlo en tus mensajes.' : 'Tu navegador bloquea el compartido directo. Copia el enlace único aquí abajo:'}
        </p>
        
        <button 
          onClick={handleCopiarEnlace}
          className="w-full py-3 bg-[#39FF14] text-black font-extrabold uppercase rounded-xl tracking-wider hover:bg-[#32e612] transition-all"
        >
          {estadoCopiado ? '¡COPIADO!' : 'COPIAR LINK DE LA APP'}
        </button>
        
        <button 
          onClick={onClose}
          className="w-full mt-3 py-2 bg-transparent text-gray-500 font-bold uppercase text-xs hover:text-white transition-colors"
        >
          Cancelar
        </button>
      </div>
    </div>
  );
};

export default EliteShareManager;
